gundata_corplot <- gundata_clean %>%
  select(-"state", -"ShallCarryLaw")
gundata_cor <- cor(gundata_corplot, method = "pearson")
col_gd <- colorRampPalette(c("#BB4444", "#EE9988", "#FFFFFF", "#77AADD", "#4477AA"))
